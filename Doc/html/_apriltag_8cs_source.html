84 private static Point FindCentroid(Point[] points)
85 {
86 // Add the first point at the end of the array.
87 int num_points = points.Length;
88 Point[] pts = new Point[num_points + 1];
89 for (int i = 0; i < points.Length; i++)
90 {
91 pts[i] = new Point(points[i].X, points[i].Y);
92 }
93 pts[num_points] = new Point(points[0].X, points[0].Y);
94
95 // Find the centroid.
96 float X = 0;
97 float Y = 0;
98 float second_factor;
99 for (int i = 0; i < num_points; i++)
100 {
101 second_factor =
102 pts[i].X * pts[i + 1].Y -
103 pts[i + 1].X * pts[i].Y;
104 X += (pts[i].X + pts[i + 1].X) * second_factor;
105 Y += (pts[i].Y + pts[i + 1].Y) * second_factor;
106 }
107
108 float area = 0;
109 for (int i = 0; i < num_points; i++)
110 {
111 area +=
112 (pts[i + 1].X - pts[i].X) *
113 (pts[i + 1].Y + pts[i].Y) / 2;
114 }
115
116 // Divide by 6 times the polygon's area.
117 X /= (6 * area);
118 Y /= (6 * area);
119
120 // If the values are negative, the polygon is
121 // oriented counterclockwise so reverse the signs.
122 if (X < 0)
123 {
124 X = -X;
125 Y = -Y;
126 }
127
128 return new Point(X, Y);
129 }
