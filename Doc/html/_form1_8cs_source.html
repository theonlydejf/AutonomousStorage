138 {
139 // Variables used often
140 Control box = sender as Control;
141 Graphics g = e.Graphics;
142
143 // If no divec is selected yet
144 if (frameProvider == null)
145 {
146 PaintError(g, "No device!", box);
147 return;

154 PaintError(g, "Waiting for first frame...", box);
155 return;
156 }
157
158 // Clone the current frame before it is desposed
159 using (Bitmap frame = frameProvider.LatestFrame.Clone() as Bitmap)
160 {
161 float sx = box.Width / (float)frame.Width;
162 float sy = box.Height / (float)frame.Height;
163 if (sx < sy)
164 sy = sx;
165 else
166 sx = sy;
167 //g.TranslateTransform(box.Width / 2 - frame.Width / 2 * sx, box.Height / 2 - frame.Height / 2 * sy);

170 // Draw the frame tothe background
171 g.DrawImage(frame, 0, 0);
172
173 Apriltag? tag0 = null;
174 // Draw all the detected apriltags
175 foreach (Apriltag tag in frameTags)
176 {
177 if (tag.ID == 0)
178 tag0 = tag;
179 // Convert OpenCV Points to System.Drawing.Point
180 Point[] pts = tag.Corners.Select(pt => new Point(pt.X, pt.Y)).ToArray();
181
182 // Draw the face of the apriltag
183 g.FillPolygon(new SolidBrush(Color.FromArgb(128, Color.Magenta)), pts);
184 // Draw the line pointg to the top of the apriltag (shows the orientation)
185 g.DrawLine(new Pen(Color.Lime, 5), tag.Center.X, tag.Center.Y, (pts[0].X + pts[1].X) / 2, (pts[0].Y + pts[1].Y) / 2);
186 // Draw the ID of the apriltag
187 g.DrawString(tag.ID.ToString(), new Font("Arail", 10, FontStyle.Bold), Brushes.Yellow, tag.Center.X, tag.Center.Y);
188 }
189
190 // If a calibration rectangle can be created
191 if (calibrationRectangle != null)
192 {
193 // Draw the outline of the rectangle
194 calibrationRectangle.Paint(g, tag0 == null ? null : tag0.Value.Center);
195 // If a tag with ID=0 is present in the frame
196 if(tag0.HasValue)
197 {
198 // Map a point to the field
199 Vector2 map = calibrationRectangle.MapPoint(tag0.Value.Center);
200 // Calculate the dimension of the scaled field
201 float width = 300;
202 float height = width / fieldSize.X * fieldSize.Y;
203
204 // Background of the field in the bottom left
205 g.FillRectangle(Brushes.White, 0, box.Height - height - 1, width, height);
206 g.DrawRectangle(Pens.Black, 0, box.Height - height - 1, width, height);
207
208 // Draws the point representing the tag0

210 }
211 }
212 }
213 }
214 // When the frame is already disposed or when any other multithread exception occures
215 catch (InvalidOperationException)
