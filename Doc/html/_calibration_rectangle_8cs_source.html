123 // Calculate the distance between
124 // point pt and the segment p1 --> p2.
125 private double FindDistanceToSegment(Point pt, Point p1, Point p2, out Point closest)
126 {
127 float dx = p2.X - p1.X;
128 float dy = p2.Y - p1.Y;
129 if ((dx == 0) && (dy == 0))
130 {
131 // It's a point not a line segment.
132 closest = p1;
133 dx = pt.X - p1.X;
134 dy = pt.Y - p1.Y;
135 return Math.Sqrt(dx * dx + dy * dy);
136 }
137
138 // Calculate the t that minimizes the distance.
139 float t = ((pt.X - p1.X) * dx + (pt.Y - p1.Y) * dy) /
140 (dx * dx + dy * dy);
141
142 closest = new Point(p1.X + t * dx, p1.Y + t * dy);
143
144 dx = pt.X - closest.X;
145 dy = pt.Y - closest.Y;
146
147 return Math.Sqrt(dx * dx + dy * dy);
148 }
